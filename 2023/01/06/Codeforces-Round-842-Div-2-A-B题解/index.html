#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2e5+5;
int p[N];
int v[N];
int ceil(int n,int k)
{
    if(n%k==0)
        return n/k;
    return n/k+1;
}
int main() {
    ios::sync_with_stdio(false);
    int T;
    cin>>T;
    while(T--)
    {
        int n,k;
        cin>>n>>k;
        for(int i=1;i<=n;i++)
        {
            cin>>p[i];
            v[p[i]]=i;
        }
        int t=1;
        for(int i=2;i<=n;i++)
        {
            if(v[i]<v[i-1])
            {
                t=i;
                break;
            }
        }
        if(t==1)
            cout<<0<<'\n';
        else
            cout<<ceil(n-t+1,k)<<'\n';
    }
    return 0;
}
